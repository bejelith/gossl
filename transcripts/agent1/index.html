Generate cost tracking CSVs from the current session's JSONL logs.

Supports both Oh My Pi (OMP) and Claude Code session formats. Auto-detects based on which session directory exists.

Output

Two files in ~/.claude/cost/YYYY/MM/DD/: - HHMMSS_sessionid_summary.csv — single row with totals - HHMMSS_sessionid_details.csv — per-interaction rows sorted by timestamp

Steps

  1. Detect session source (OMP or Claude Code) and find the current session JSONL

import json, os, csv, glob
from datetime import datetime

PRICING = {
    "claude-opus-4-6": {"input": 5/1e6, "output": 25/1e6, "cache_write": 6.25/1e6, "cache_read": 0.50/1e6},
    "claude-sonnet-4-6": {"input": 3/1e6, "output": 15/1e6, "cache_write": 3.75/1e6, "cache_read": 0.30/1e6},
    "claude-haiku-4-5": {"input": 1/1e6, "output": 5/1e6, "cache_write": 1.25/1e6, "cache_read": 0.10/1e6},
}

def get_pricing(model):
    for key, p in PRICING.items():
        if key in model:
            return p
    return PRICING["claude-opus-4-6"]

def detect_format(filepath):
    """Read first line to detect OMP vs Claude Code format."""
    with open(filepath) as f:
        first = f.readline()
        try:
            entry = json.loads(first)
            if entry.get("type") == "session" and entry.get("version", 0) >= 3:
                return "omp"
        except:
            pass
    return "claude"

def parse_omp_jsonl(filepath, source):
    """Parse OMP session JSONL. Messages use camelCase keys and nested message object."""
    rows = []
    with open(filepath) as f:
        for line in f:
            try:
                entry = json.loads(line)
            except:
                continue
            if entry.get("type") != "message":
                continue
            msg = entry.get("message", {})
            if msg.get("role") != "assistant":
                continue
            stop = msg.get("stopReason", "")
            if not stop or stop == "aborted":
                continue
            usage = msg.get("usage", {})
            model = msg.get("model", "unknown")
            input_tokens = usage.get("input", 0)
            output_tokens = usage.get("output", 0)
            cache_write = usage.get("cacheWrite", 0)
            cache_read = usage.get("cacheRead", 0)
            p = get_pricing(model)
            cost = (input_tokens * p["input"] + output_tokens * p["output"] +
                    cache_write * p["cache_write"] + cache_read * p["cache_read"])
            rows.append({
                "timestamp": entry.get("timestamp", ""),
                "source": source,
                "model": model,
                "provider": msg.get("provider", ""),
                "input_tokens": input_tokens,
                "output_tokens": output_tokens,
                "cache_write_tokens": cache_write,
                "cache_read_tokens": cache_read,
                "cost_usd": round(cost, 6),
                "stop_reason": stop,
                "duration_ms": msg.get("duration", 0),
                "ttft_ms": msg.get("ttft", 0),
            })
    return rows

def parse_claude_jsonl(filepath, source):
    """Parse Claude Code session JSONL. Messages use snake_case keys at top level."""
    rows = []
    with open(filepath) as f:
        for line in f:
            try:
                entry = json.loads(line)
            except:
                continue
            if entry.get("type") != "assistant":
                continue
            msg = entry.get("message", {})
            if not msg.get("stop_reason"):
                continue
            usage = msg.get("usage", {})
            model = msg.get("model", "unknown")
            input_tokens = usage.get("input_tokens", 0)
            output_tokens = usage.get("output_tokens", 0)
            cache_write = usage.get("cache_creation_input_tokens", 0)
            cache_read = usage.get("cache_read_input_tokens", 0)
            p = get_pricing(model)
            cost = (input_tokens * p["input"] + output_tokens * p["output"] +
                    cache_write * p["cache_write"] + cache_read * p["cache_read"])
            rows.append({
                "timestamp": entry.get("timestamp", ""),
                "source": source,
                "model": model,
                "provider": "",
                "input_tokens": input_tokens,
                "output_tokens": output_tokens,
                "cache_write_tokens": cache_write,
                "cache_read_tokens": cache_read,
                "cost_usd": round(cost, 6),
                "stop_reason": msg.get("stop_reason", ""),
                "duration_ms": 0,
                "ttft_ms": 0,
            })
    return rows

def find_subagent_jsonls(session_dir, session_id, fmt):
    """Find subagent JSONL files for a session."""
    if fmt == "omp":
        # OMP: subagent logs are *.jsonl inside {session_dir}/{session_id}/
        sub_dir = os.path.join(session_dir, session_id)
    else:
        # Claude Code: subagent logs are in {session_dir}/{session_id}/subagents/
        sub_dir = os.path.join(session_dir, session_id, "subagents")
    if not os.path.isdir(sub_dir):
        return []
    return [
        (os.path.join(sub_dir, f), f.replace(".jsonl", ""))
        for f in os.listdir(sub_dir)
        if f.endswith(".jsonl")
    ]

def save_cost(session_id, session_dir):
    main_jsonl = os.path.join(session_dir, f"{session_id}.jsonl")
    fmt = detect_format(main_jsonl)
    parse = parse_omp_jsonl if fmt == "omp" else parse_claude_jsonl

    now = datetime.now()
    cost_dir = os.path.expanduser(now.strftime("~/.claude/cost/%Y/%m/%d"))
    os.makedirs(cost_dir, exist_ok=True)

    all_rows = parse(main_jsonl, "main")
    for sub_path, agent_id in find_subagent_jsonls(session_dir, session_id, fmt):
        all_rows.extend(parse(sub_path, agent_id))

    all_rows.sort(key=lambda r: r["timestamp"])

    models = sorted(set(r["model"] for r in all_rows))
    providers = sorted(set(r["provider"] for r in all_rows if r["provider"]))
    total_input = sum(r["input_tokens"] for r in all_rows)
    total_output = sum(r["output_tokens"] for r in all_rows)
    total_cache_write = sum(r["cache_write_tokens"] for r in all_rows)
    total_cache_read = sum(r["cache_read_tokens"] for r in all_rows)
    total_cost = sum(r["cost_usd"] for r in all_rows)
    total_duration = sum(r["duration_ms"] for r in all_rows)

    ts = now.strftime("%H%M%S")

    summary_path = os.path.join(cost_dir, f"{ts}_{session_id}_summary.csv")
    with open(summary_path, "w", newline="") as f:
        w = csv.writer(f)
        w.writerow(["session_id", "format", "providers", "models",
                     "total_input_tokens", "total_output_tokens",
                     "total_cache_write_tokens", "total_cache_read_tokens",
                     "total_cost_usd", "num_interactions", "total_duration_ms"])
        w.writerow([session_id, fmt, ";".join(providers), ";".join(models),
                     total_input, total_output, total_cache_write, total_cache_read,
                     round(total_cost, 4), len(all_rows), total_duration])

    details_path = os.path.join(cost_dir, f"{ts}_{session_id}_details.csv")
    fields = ["timestamp", "source", "model", "provider",
              "input_tokens", "output_tokens", "cache_write_tokens", "cache_read_tokens",
              "cost_usd", "stop_reason", "duration_ms", "ttft_ms"]
    with open(details_path, "w", newline="") as f:
        w = csv.DictWriter(f, fieldnames=fields)
        w.writeheader()
        w.writerows(all_rows)

    main_count = len([r for r in all_rows if r["source"] == "main"])
    sub_count = len(all_rows) - main_count
    print(f"Format: {fmt}")
    if providers:
        print(f"Provider: {', '.join(providers)}")
    print(f"Interactions: {len(all_rows)} ({main_count} main, {sub_count} subagent)")
    print(f"Tokens: {total_input:,} in / {total_output:,} out / {total_cache_write:,} cache-w / {total_cache_read:,} cache-r")
    print(f"Total cost: ${round(total_cost, 4)}")
    if total_duration > 0:
        print(f"Total API time: {total_duration / 1000:.1f}s")
    print(f"Summary: {summary_path}")
    print(f"Details: {details_path}")
