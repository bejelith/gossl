> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement the SQLite storage layer (certstore + keystore) with full test coverage.

**Architecture:** Two store implementations behind interfaces defined in `internal/ca/`. `certstore` handles CA/intermediate/cert CRUD in SQLite. `keystore` is a KV API for private keys, also SQLite-backed. Schema versioning via `PRAGMA user_version`. Hand-written SQL queries for now; sqlc integration deferred until the schema stabilizes.

**Tech Stack:** Go, SQLite (modernc.org/sqlite), Bazel

```go
// Package ca provides certificate authority operations.
package ca

import (
	"context"
	"time"
)

// CACert represents a root CA stored in the database.
type CACert struct {
	ID         int64
	CommonName string
	Serial     string
	KeyAlgo    string
	CertPEM    string
	NotBefore  time.Time
	NotAfter   time.Time
	CreatedAt  time.Time
}

// IntermediateCert represents an intermediate CA.
type IntermediateCert struct {
	ID         int64
	CAID       int64
	CommonName string
	Serial     string
	KeyAlgo    string
	CertPEM    string
	NotBefore  time.Time
	NotAfter   time.Time
	CreatedAt  time.Time
}

// Cert represents an issued certificate.
type Cert struct {
	ID             int64
	IntermediateID int64
	CommonName     string
	Serial         string
	KeyAlgo        string
	CertPEM        string
	NotBefore      time.Time
	NotAfter       time.Time
	RevokedAt      *time.Time
	CreatedAt      time.Time
}

// CertStore manages certificates in persistent storage.
type CertStore interface {
	CreateCA(ctx context.Context, c CACert) (CACert, error)
	GetCA(ctx context.Context) (CACert, error)

	CreateIntermediate(ctx context.Context, c IntermediateCert) (IntermediateCert, error)
	GetIntermediate(ctx context.Context, id int64) (IntermediateCert, error)
	GetIntermediateBySerial(ctx context.Context, serial string) (IntermediateCert, error)
	GetIntermediateByCN(ctx context.Context, cn string) (IntermediateCert, error)
	ListIntermediates(ctx context.Context) ([]IntermediateCert, error)

	CreateCert(ctx context.Context, c Cert) (Cert, error)
	GetCert(ctx context.Context, id int64) (Cert, error)
	GetCertBySerial(ctx context.Context, serial string) (Cert, error)
	GetCertByCN(ctx context.Context, cn string) (Cert, error)
	ListCerts(ctx context.Context) ([]Cert, error)
	ListCertsByIntermediate(ctx context.Context, intermediateID int64) ([]Cert, error)
	RevokeCert(ctx context.Context, serial string, revokedAt time.Time) error
}

// KeyStore manages private keys in persistent storage.
type KeyStore interface {
	Store(ctx context.Context, id string, keyPEM []byte) error
	Load(ctx context.Context, id string) ([]byte, error)
	Delete(ctx context.Context, id string) error
}
```

// Package store provides SQLite-backed storage implementations.
package store

import (
	"context"
	"database/sql"
	_ "embed"
	"fmt"

	_ "modernc.org/sqlite"
)

//go:embed schema.sql
var schemaSQL string

// SchemaVersion is the current database schema version.
const SchemaVersion = 1

// DB wraps a sql.DB connection with schema management.
type DB struct {
	db *sql.DB
}

// OpenDB opens a SQLite database and ensures the schema is initialized.
// Use ":memory:" for in-memory databases (testing).
func OpenDB(ctx context.Context, path string) (*DB, error) {
	sqlDB, err := sql.Open("sqlite", path)
	if err != nil {
		return nil, fmt.Errorf("opening database: %w", err)
	}

	if err := sqlDB.PingContext(ctx); err != nil {
		sqlDB.Close()
		return nil, fmt.Errorf("connecting to database: %w", err)
	}

	if _, err := sqlDB.ExecContext(ctx, "PRAGMA foreign_keys = ON"); err != nil {
		sqlDB.Close()
		return nil, fmt.Errorf("enabling foreign keys: %w", err)
	}

	var version int
	if err := sqlDB.QueryRowContext(ctx, "PRAGMA user_version").Scan(&version); err != nil {
		sqlDB.Close()
		return nil, fmt.Errorf("reading schema version: %w", err)
	}

	if version == 0 {
		if _, err := sqlDB.ExecContext(ctx, schemaSQL); err != nil {
			sqlDB.Close()
			return nil, fmt.Errorf("initializing schema: %w", err)
		}
		if _, err := sqlDB.ExecContext(ctx, fmt.Sprintf("PRAGMA user_version = %d", SchemaVersion)); err != nil {
			sqlDB.Close()
			return nil, fmt.Errorf("setting schema version: %w", err)
		}
	} else if version != SchemaVersion {
		sqlDB.Close()
		return nil, fmt.Errorf("unsupported schema version %d (expected %d)", version, SchemaVersion)
	}

	return &DB{db: sqlDB}, nil
}

// Close closes the underlying database connection.
func (d *DB) Close() error {
	return d.db.Close()
}

// SqlDB returns the underlying *sql.DB for query execution.
func (d *DB) SqlDB() *sql.DB {
	return d.db
}
```

package store

import (
	"context"
	"database/sql"
	"fmt"
)

// SQLiteKeyStore implements ca.KeyStore backed by SQLite.
type SQLiteKeyStore struct {
	db *DB
}

// NewKeyStore creates a new SQLite-backed KeyStore.
func NewKeyStore(db *DB) *SQLiteKeyStore {
	return &SQLiteKeyStore{db: db}
}

// Store saves a private key PEM by ID.
func (ks *SQLiteKeyStore) Store(ctx context.Context, id string, keyPEM []byte) error {
	_, err := ks.db.SqlDB().ExecContext(ctx, "INSERT INTO keystore (id, key_pem) VALUES (?, ?)", id, keyPEM)
	if err != nil {
		return fmt.Errorf("storing key %s: %w", id, err)
	}
	return nil
}

// Load retrieves a private key PEM by ID.
func (ks *SQLiteKeyStore) Load(ctx context.Context, id string) ([]byte, error) {
	var keyPEM []byte
	err := ks.db.SqlDB().QueryRowContext(ctx, "SELECT key_pem FROM keystore WHERE id = ?", id).Scan(&keyPEM)
	if err == sql.ErrNoRows {
		return nil, fmt.Errorf("key %s not found", id)
	}
	if err != nil {
		return nil, fmt.Errorf("loading key %s: %w", id, err)
	}
	return keyPEM, nil
}

// Delete removes a private key by ID.
func (ks *SQLiteKeyStore) Delete(ctx context.Context, id string) error {
	res, err := ks.db.SqlDB().ExecContext(ctx, "DELETE FROM keystore WHERE id = ?", id)
	if err != nil {
		return fmt.Errorf("deleting key %s: %w", id, err)
	}
	rows, err := res.RowsAffected()
	if err != nil {
		return fmt.Errorf("deleting key %s: %w", id, err)
	}
	if rows == 0 {
		return fmt.Errorf("key %s not found", id)
	}
	return nil
}
```

package store

import (
	"context"
	"database/sql"
	"fmt"
	"time"

	"github.com/scaruso/gossl/internal/ca"
)

// SQLiteCertStore implements ca.CertStore backed by SQLite.
type SQLiteCertStore struct {
	db *DB
}

// NewCertStore creates a new SQLite-backed CertStore.
func NewCertStore(db *DB) *SQLiteCertStore {
	return &SQLiteCertStore{db: db}
}

func (s *SQLiteCertStore) CreateCA(ctx context.Context, c ca.CACert) (ca.CACert, error) {
	_, err := s.db.SqlDB().ExecContext(ctx,
		`INSERT INTO ca (id, common_name, serial, key_algo, cert_pem, not_before, not_after, created_at)
		 VALUES (1, ?, ?, ?, ?, ?, ?, ?)`,
		c.CommonName, c.Serial, c.KeyAlgo, c.CertPEM, c.NotBefore, c.NotAfter, c.CreatedAt,
	)
	if err != nil {
		return ca.CACert{}, fmt.Errorf("creating CA: %w", err)
	}
	c.ID = 1
	return c, nil
}

func (s *SQLiteCertStore) GetCA(ctx context.Context) (ca.CACert, error) {
	row := s.db.SqlDB().QueryRowContext(ctx,
		`SELECT id, common_name, serial, key_algo, cert_pem, not_before, not_after, created_at FROM ca WHERE id = 1`)
	var c ca.CACert
	err := row.Scan(&c.ID, &c.CommonName, &c.Serial, &c.KeyAlgo, &c.CertPEM, &c.NotBefore, &c.NotAfter, &c.CreatedAt)
	if err == sql.ErrNoRows {
		return ca.CACert{}, fmt.Errorf("no CA found")
	}
	if err != nil {
		return ca.CACert{}, fmt.Errorf("getting CA: %w", err)
	}
	return c, nil
}

func (s *SQLiteCertStore) CreateIntermediate(ctx context.Context, c ca.IntermediateCert) (ca.IntermediateCert, error) {
	res, err := s.db.SqlDB().ExecContext(ctx,
		`INSERT INTO intermediate (ca_id, common_name, serial, key_algo, cert_pem, not_before, not_after, created_at)
		 VALUES (?, ?, ?, ?, ?, ?, ?, ?)`,
		c.CAID, c.CommonName, c.Serial, c.KeyAlgo, c.CertPEM, c.NotBefore, c.NotAfter, c.CreatedAt,
	)
	if err != nil {
		return ca.IntermediateCert{}, fmt.Errorf("creating intermediate: %w", err)
	}
	id, _ := res.LastInsertId()
	c.ID = id
	return c, nil
}

func (s *SQLiteCertStore) GetIntermediate(ctx context.Context, id int64) (ca.IntermediateCert, error) {
	row := s.db.SqlDB().QueryRowContext(ctx,
		`SELECT id, ca_id, common_name, serial, key_algo, cert_pem, not_before, not_after, created_at
		 FROM intermediate WHERE id = ?`, id)
	return scanIntermediate(row)
}

func (s *SQLiteCertStore) GetIntermediateBySerial(ctx context.Context, serial string) (ca.IntermediateCert, error) {
	row := s.db.SqlDB().QueryRowContext(ctx,
		`SELECT id, ca_id, common_name, serial, key_algo, cert_pem, not_before, not_after, created_at
		 FROM intermediate WHERE serial = ?`, serial)
	return scanIntermediate(row)
}

func (s *SQLiteCertStore) GetIntermediateByCN(ctx context.Context, cn string) (ca.IntermediateCert, error) {
	row := s.db.SqlDB().QueryRowContext(ctx,
		`SELECT id, ca_id, common_name, serial, key_algo, cert_pem, not_before, not_after, created_at
		 FROM intermediate WHERE common_name = ?`, cn)
	return scanIntermediate(row)
}

func (s *SQLiteCertStore) ListIntermediates(ctx context.Context) ([]ca.IntermediateCert, error) {
	rows, err := s.db.SqlDB().QueryContext(ctx,
		`SELECT id, ca_id, common_name, serial, key_algo, cert_pem, not_before, not_after, created_at
		 FROM intermediate ORDER BY created_at DESC`)
	if err != nil {
		return nil, fmt.Errorf("listing intermediates: %w", err)
	}
	defer rows.Close()

	var result []ca.IntermediateCert
	for rows.Next() {
		var c ca.IntermediateCert
		if err := rows.Scan(&c.ID, &c.CAID, &c.CommonName, &c.Serial, &c.KeyAlgo, &c.CertPEM, &c.NotBefore, &c.NotAfter, &c.CreatedAt); err != nil {
			return nil, fmt.Errorf("scanning intermediate: %w", err)
		}
		result = append(result, c)
	}
	return result, rows.Err()
}

func (s *SQLiteCertStore) CreateCert(ctx context.Context, c ca.Cert) (ca.Cert, error) {
	res, err := s.db.SqlDB().ExecContext(ctx,
		`INSERT INTO cert (intermediate_id, common_name, serial, key_algo, cert_pem, not_before, not_after, created_at)
		 VALUES (?, ?, ?, ?, ?, ?, ?, ?)`,
		c.IntermediateID, c.CommonName, c.Serial, c.KeyAlgo, c.CertPEM, c.NotBefore, c.NotAfter, c.CreatedAt,
	)
	if err != nil {
		return ca.Cert{}, fmt.Errorf("creating cert: %w", err)
	}
	id, _ := res.LastInsertId()
	c.ID = id
	return c, nil
}

func (s *SQLiteCertStore) GetCert(ctx context.Context, id int64) (ca.Cert, error) {
	row := s.db.SqlDB().QueryRowContext(ctx,
		`SELECT id, intermediate_id, common_name, serial, key_algo, cert_pem, not_before, not_after, revoked_at, created_at
		 FROM cert WHERE id = ?`, id)
	return scanCert(row)
}

func (s *SQLiteCertStore) GetCertBySerial(ctx context.Context, serial string) (ca.Cert, error) {
	row := s.db.SqlDB().QueryRowContext(ctx,
		`SELECT id, intermediate_id, common_name, serial, key_algo, cert_pem, not_before, not_after, revoked_at, created_at
		 FROM cert WHERE serial = ?`, serial)
	return scanCert(row)
}

func (s *SQLiteCertStore) GetCertByCN(ctx context.Context, cn string) (ca.Cert, error) {
	row := s.db.SqlDB().QueryRowContext(ctx,
		`SELECT id, intermediate_id, common_name, serial, key_algo, cert_pem, not_before, not_after, revoked_at, created_at
		 FROM cert WHERE common_name = ?`, cn)
	return scanCert(row)
}

func (s *SQLiteCertStore) ListCerts(ctx context.Context) ([]ca.Cert, error) {
	return s.queryCerts(ctx,
		`SELECT id, intermediate_id, common_name, serial, key_algo, cert_pem, not_before, not_after, revoked_at, created_at
		 FROM cert ORDER BY created_at DESC`)
}

func (s *SQLiteCertStore) ListCertsByIntermediate(ctx context.Context, intermediateID int64) ([]ca.Cert, error) {
	return s.queryCerts(ctx,
		`SELECT id, intermediate_id, common_name, serial, key_algo, cert_pem, not_before, not_after, revoked_at, created_at
		 FROM cert WHERE intermediate_id = ? ORDER BY created_at DESC`, intermediateID)
}

func (s *SQLiteCertStore) RevokeCert(ctx context.Context, serial string, revokedAt time.Time) error {
	res, err := s.db.SqlDB().ExecContext(ctx, `UPDATE cert SET revoked_at = ? WHERE serial = ?`, revokedAt, serial)
	if err != nil {
		return fmt.Errorf("revoking cert %s: %w", serial, err)
	}
	rows, err := res.RowsAffected()
	if err != nil {
		return fmt.Errorf("revoking cert %s: %w", serial, err)
	}
	if rows == 0 {
		return fmt.Errorf("cert %s not found", serial)
	}
	return nil
}

// Helpers

func (s *SQLiteCertStore) queryCerts(ctx context.Context, query string, args ...any) ([]ca.Cert, error) {
	rows, err := s.db.SqlDB().QueryContext(ctx, query, args...)
	if err != nil {
		return nil, fmt.Errorf("listing certs: %w", err)
	}
	defer rows.Close()

	var result []ca.Cert
	for rows.Next() {
		var c ca.Cert
		if err := rows.Scan(&c.ID, &c.IntermediateID, &c.CommonName, &c.Serial, &c.KeyAlgo, &c.CertPEM, &c.NotBefore, &c.NotAfter, &c.RevokedAt, &c.CreatedAt); err != nil {
			return nil, fmt.Errorf("scanning cert: %w", err)
		}
		result = append(result, c)
	}
	return result, rows.Err()
}

func scanIntermediate(row *sql.Row) (ca.IntermediateCert, error) {
	var c ca.IntermediateCert
	err := row.Scan(&c.ID, &c.CAID, &c.CommonName, &c.Serial, &c.KeyAlgo, &c.CertPEM, &c.NotBefore, &c.NotAfter, &c.CreatedAt)
	if err == sql.ErrNoRows {
		return ca.IntermediateCert{}, fmt.Errorf("intermediate not found")
	}
	if err != nil {
		return ca.IntermediateCert{}, fmt.Errorf("getting intermediate: %w", err)
	}
	return c, nil
}

func scanCert(row *sql.Row) (ca.Cert, error) {
	var c ca.Cert
	err := row.Scan(&c.ID, &c.IntermediateID, &c.CommonName, &c.Serial, &c.KeyAlgo, &c.CertPEM, &c.NotBefore, &c.NotAfter, &c.RevokedAt, &c.CreatedAt)
	if err == sql.ErrNoRows {
		return ca.Cert{}, fmt.Errorf("cert not found")
	}
	if err != nil {
		return ca.Cert{}, fmt.Errorf("getting cert: %w", err)
	}
	return c, nil
}
